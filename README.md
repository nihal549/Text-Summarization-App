# minor_project
backend:
To run first do
cd mini_project/
pip install -r requirements.txt
uvicorn app:app --reload
press CTRL+C to quit

frontend:
cd mini_project/ui
npm init --y
npm i express
npm i body-parser
npm i cors

node version:16.18.0

#To run the project
cd mini_project/
uvicorn app:app
cd to ui/js
node server.js
click on index.html Or use live server
press Ctrl+c to quit both servers
